# Scholario Insomnia Operations - Testing Manual
# ===============================================
# HOW TO USE IN INSOMNIA:
# 1. Create a new "HTTP Request".
# 2. Set Method to "POST" and URL to "http://localhost:8090/graphql".
# 3. Select the "Body" tab and choose "GraphQL".
# 4. Copy the "QUERY" block below into the "Query" field in Insomnia.
# 5. Copy the "VARIABLES" block below into the "Variables" JSON field in Insomnia.
# 6. For Auth: Go to "Auth" tab -> "Bearer Token" and paste your accessToken.

################################################################################

# [1.1] Register a New User
# QUERY:
mutation RegisterUser($input: UserInput!) {
  registerUser(input: $input) {
    id
    username
    email
    fullName
    roles
  }
}
# VARIABLES:
{
  "input": {
    "username": "faculty_jane",
    "password": "Password123!",
    "email": "jane@scholario.edu",
    "fullName": "Jane Doe",
    "roles": "FACULTY"
  }
}

# [1.2] Login to get JWT Token
# QUERY:
mutation Login($input: LoginInput!) {
  login(input: $input) {
    accessToken
    refreshToken
    tokenType
    expiresIn
    user {
      id
      username
      roles
    }
  }
}
# VARIABLES:
{
  "input": {
    "username": "faculty_jane",
    "password": "Password123!"
  }
}
