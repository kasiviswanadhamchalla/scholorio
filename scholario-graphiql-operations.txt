# Scholario GraphiQL Operations - Step-by-Step Testing Manual
# ==========================================================
# HOW TO USE:
# 1. Copy the GraphQL code into the "Query" pane of GraphiQL.
# 2. Copy the JSON from the "# Variables" comment into the "Query Variables" pane (bottom left).
# 3. For authenticated requests, copy the 'accessToken' from the Login response and add it to the 
#    "HTTP HEADERS" pane as: { "Authorization": "Bearer YOUR_TOKEN_HERE" }

################################################################################
# 1. AUTHENTICATION & REGISTRATION
################################################################################

# [1.1] Register a New User
# Variables:
# {
#   "input": {
#     "username": "faculty_jane",
#     "password": "Password123!",
#     "email": "jane@scholario.edu",
#     "fullName": "Jane Doe",
#     "roles": "FACULTY"
#   }
# }
mutation RegisterUser($input: UserInput!) {
  registerUser(input: $input) {
    id
    username
    email
    fullName
    roles
  }
}

# [1.2] Login to get your JWT Token
# Variables:
# {
#   "input": {
#     "username": "faculty_jane",
#     "password": "Password123!"
#   }
# }
mutation Login($input: LoginInput!) {
  login(input: $input) {
    accessToken
    refreshToken
    tokenType
    expiresIn
    user {
      id
      username
      roles
    }
  }
}

# [1.3] Refresh Token
# Variables:
# {
#   "refreshToken": "PASTE_YOUR_REFRESH_TOKEN_HERE"
# }
mutation RefreshToken($refreshToken: String!) {
  refreshToken(refreshToken: $refreshToken) {
    accessToken
    refreshToken
    tokenType
  }
}

# [1.4] Validate Token
# Variables:
# {
#   "token": "PASTE_YOUR_ACCESS_TOKEN_HERE"
# }
query ValidateToken($token: String!) {
  validateToken(token: $token) {
    valid
    username
    roles
    expiresAt
  }
}

# [1.5] Logout
mutation Logout {
  logout
}
